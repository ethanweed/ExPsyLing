# McElreath globe exercise
# re-envisioned as the SantaBall exercise




import matplotlib.pyplot as plt
import numpy as np
import scipy

number_red = 0
throws = 0


prob = np.linspace(0, 1, num=10)
y = scipy.stats.binom.pmf(number_red, throws, prob)     # number of successes, number of trials, probability of a single success

fig, ax = plt.subplots()
ax.plot(prob, y)

ax.set(xlabel='probability', ylabel='binomial liklihood',
       title='Estimate of proportion of red on the Santa Ball')
ax.grid()

Two Ways to Quantify Inductive Inferences

# Frequentist test

from scipy.stats import binom_test
pvalue = binom_test(x = number_red, n = throws, p = 0.5, alternative = 'two-sided')
print("Probability of measuring values as extreme as our data if the null hypothesis is true, aka the p-value:", round(pvalue, 3))
# Bayes factor

import pingouin as pg

bf_alt = pg.bayesfactor_binom(k = number_red, n = throws, p = 0.5)
bf_null = 1/bf_alt

print("Strength of evidence in favor of the null hypothesis:", round(bf_null, 3))
print("Strength of evidence in favor of the alternate hypothesis:", round(bf_alt,3))
